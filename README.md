#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--){
	    string s;
	    cin>>s;
	    int n = s.length();
	    for(int i = 0; i < n/2; i++){
	        swap(s[i],s[n-1-i]);
	    }
	    cout<<s<<endl;
	}
	return 0;
}
#python
class Solution:
    def reverseWord(s):
        # for reversing the array we will traverse it from 0th index to the mid index
        # and keep swapping the ith index value with (len(s)-i)th index value
        l=len(s)
        
        for i in range(len(s)//2):
            s[i],s[l-i-1] = s[l-i-1],s[i]
            
        return(s)
